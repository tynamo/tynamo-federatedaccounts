044
045        private Logger logger;
046
047        private FederatedAccountService federatedAccountService;
048
049        public Pac4jFederatedRealm(Logger logger, FederatedAccountService federatedAccountService) {
050
051                super(new MemoryConstrainedCacheManager());
052                this.federatedAccountService = federatedAccountService;
053                this.logger = logger;
054
055                setName(pac4j_.name());
056                setAuthenticationTokenClass(Pac4jAuthenticationToken.class);
057        }
058
059        @Override
060        protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken)
061                throws AuthenticationException {
062
063                Pac4jAuthenticationToken token = (Pac4jAuthenticationToken) authenticationToken;
064                UserProfile profile = (UserProfile) token.getPrincipal();
065                String realmName = pac4j_.name() + profile.getClass().getSimpleName().toLowerCase();
066                if (realmName.endsWith("profile")) realmName = realmName.substring(0, realmName.lastIndexOf("profile"));
067                /*
068                 * OAuthRequest request = new OAuthRequest(Verb.GET, "https://www.google.com/m8/feeds/contacts/default/full");
069                 * service.signRequest(token.getToken(), request); request.send().getBody();
070                 */
071
072                // Null username is invalid, throw an exception if so - indicates that user hasn't granted the right
073                // permissions (and/or we haven't asked for it)
074                /*
075                 * if (facebookUser == null) throw new AccountException("Null Facebook user is not allowed by this realm.");
076                 * 
077                 * String principalValue = null; switch (principalProperty) { case id: principalValue = facebookUser.getId(); break; case email:
078                 * principalValue = facebookUser.getEmail(); break; case name: principalValue = facebookUser.getName(); break; }
079                 */
080
081                return federatedAccountService.federate(realmName, profile.getId(), authenticationToken, profile);
082                // SimplePrincipalCollection principalCollection = new SimplePrincipalCollection(authenticationToken.getPrincipal(),
083                // pac4j.name());
084                // principalCollection.add(authenticationToken, pac4j.name());
085                // return new SimpleAuthenticationInfo(principalCollection, authenticationToken.getCredentials());
086
087        }
088}
