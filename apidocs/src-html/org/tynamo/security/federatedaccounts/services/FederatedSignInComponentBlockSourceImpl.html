001package org.tynamo.security.federatedaccounts.services;
002
003import java.util.Collection;
004import java.util.Map;
005
006import org.apache.tapestry5.Block;
007import org.apache.tapestry5.internal.services.RequestPageCache;
008import org.apache.tapestry5.internal.structure.Page;
009import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
010import org.slf4j.Logger;
011
012public class FederatedSignInComponentBlockSourceImpl implements FederatedSignInComponentBlockSource {
013        private final Map<String, FederatedSignInComponentContribution> signInComponentMap = CollectionFactory
014                .newCaseInsensitiveMap();
015
016        private Logger logger;
017
018        private RequestPageCache pageCache;
019
020        public FederatedSignInComponentBlockSourceImpl(Logger logger, RequestPageCache pageCache,
021                Collection<FederatedSignInComponentContribution> configuration) {
022                this.logger = logger;
023                this.pageCache = pageCache;
024                for (FederatedSignInComponentContribution contribution : configuration)
025                        signInComponentMap.put(contribution.getKey(), contribution);
026        }
027
028        @Override
029        public Block toBlock(String key) {
030                FederatedSignInComponentContribution contribution = signInComponentMap.get(key);
031                if (contribution == null) return null;
032
033                Page page = pageCache.get(contribution.getPageName());
034
035                return page.getRootElement().getBlock(contribution.getKey() + "_small"); // contribution.getBlockId());
036        }
037
038}
