001package org.tynamo.security.federatedaccounts.services;
002
003import java.util.HashSet;
004import java.util.Map;
005import java.util.Set;
006
007import org.apache.tapestry5.Block;
008
009public class FederatedSignInOptionsImpl implements FederatedSignInOptions {
010        private Set<Block> primaryBlocks;
011        private Set<Block> secondaryBlocks;
012
013        public FederatedSignInOptionsImpl(FederatedSignInComponentBlockSource componentBlockSource,
014                Map<String, OptionType> configuration) {
015                primaryBlocks = new HashSet<Block>();
016                secondaryBlocks = new HashSet<Block>();
017
018                for (Map.Entry<String, OptionType> entry : configuration.entrySet()) {
019                        if (OptionType.primary.equals(entry.getValue())) primaryBlocks.add(componentBlockSource.toBlock(entry.getKey()));
020                        else if (OptionType.secondary.equals(entry.getValue()))
021                                secondaryBlocks.add(componentBlockSource.toBlock(entry.getKey()));
022                }
023
024        }
025
026        public Set<Block> getPrimaryBlocks() {
027                return primaryBlocks;
028        }
029
030        public Set<Block> getSecondaryBlocks() {
031                return secondaryBlocks;
032        }
033}
