70  	protected void assertLoginPage(HtmlPage page) {
71  		assertNotNull(page.getElementById("tynamoLogin"), "Page doesn't contain login field. Not a login page.");
72  		assertEquals("password", getAttribute("tynamoPassword", "type"),
73  			"Page doesn't contain password field. Not a login page.");
74  		assertEquals("checkbox", getAttribute("tynamoRememberMe", "type"),
75  			"Page doesn't contain rememberMe field. Not a login page.");
76  
77  		assertNotNull(page.getElementById("tynamoEnter"),
78  			"Page doesn't contain login form submit button. Not a login page.");
79  	}
