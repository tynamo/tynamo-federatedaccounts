1   package org.tynamo.security.federatedaccounts.services;

3   import java.util.HashSet;
4   import java.util.Map;
5   import java.util.Set;
6   
7   import org.apache.tapestry5.Block;
8   
9   public class FederatedSignInOptionsImpl implements FederatedSignInOptions {
10  	private Set<Block> primaryBlocks;
11  	private Set<Block> secondaryBlocks;
12  
13  	public FederatedSignInOptionsImpl(FederatedSignInComponentBlockSource componentBlockSource,
14  		Map<String, OptionType> configuration) {
15  		primaryBlocks = new HashSet<Block>();
16  		secondaryBlocks = new HashSet<Block>();
17  
18  		for (Map.Entry<String, OptionType> entry : configuration.entrySet()) {
19  			if (OptionType.primary.equals(entry.getValue())) primaryBlocks.add(componentBlockSource.toBlock(entry.getKey()));
20  			else if (OptionType.secondary.equals(entry.getValue()))
21  				secondaryBlocks.add(componentBlockSource.toBlock(entry.getKey()));
22  		}
23  
24  	}
25  
26  	public Set<Block> getPrimaryBlocks() {
27  		return primaryBlocks;
28  	}
29  
30  	public Set<Block> getSecondaryBlocks() {
31  		return secondaryBlocks;
32  	}
33  }
