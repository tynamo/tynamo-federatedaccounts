10  public class RollingToken implements AuthenticationToken, HostAuthenticationToken, RememberMeAuthenticationToken {
11  	public static final String TOKEN_NAME = "rollingToken";
12  
13  	private Object principal;
14  	private String rollingToken;
15  	private String hostAddress;
16  
17  	public RollingToken(Object principal, String rollingToken, String remoteAddress) {
18  		this.principal = principal;
19  		this.rollingToken = rollingToken;
20  		hostAddress = remoteAddress;
21  	}
22  
23  	/**
24  	 * 
25  	 */
26  	private static final long serialVersionUID = -3054718345404435729L;
27  
28  	public Object getCredentials() {
29  		return rollingToken;
30  	}
31  
32  	public Object getPrincipal() {
33  		return principal;
34  	}
35  
36  	public boolean isRememberMe() {
37  		// Always return true, otherwise rememberMe token is erased when this authentication is used
38  		return true;
39  	}
40  
41  	public String getHost() {
42  		return hostAddress;
43  	}
44  
45  	// TODO the implementation is almost the same as in tapestry-security-jpa's SecureEntityManager.getConfiguredPrincipal
46  	// is there a possibility to refactor this to some common class?
47  	public static Object getConfiguredPrincipal(String realmName, Class<?> principalType,
48  		PrincipalCollection allPrincipals) {
49  		if (!realmName.isEmpty()) {
50  			Collection<?> principals = allPrincipals.fromRealm(realmName);
51  			if (principalType == null) principals.iterator().next();
52  			else {
53  				for (Object availablePrincipal : principals)
54  					if (availablePrincipal.getClass().isAssignableFrom(principalType)) { return availablePrincipal; }
55  			}
56  		}
57  		if (principalType != null) {
58  			Object principal = allPrincipals.oneByType(principalType);
59  			if (principal == null)
60  				throw new NullPointerException("Subject is required to have a configured principal of type '" + principalType);
61  			return principal;
62  		}
63  		return allPrincipals.getPrimaryPrincipal();
64  	}

66  }
